#Разделитель - символ пробела
#Желательно в конце ставить разделитель, на случай если в сокет попадет чтото лишнее, без разделителя в конце оно будет воспринято как продолжение последнего аргумента.
#Установка [pckname] - имя пакета [pckpath] путь к пакету
#Флаги установки: u - полный путь к пакету заместо его имени; f - fakeinstall, установка без фактического копированиф файлов; d - не устанавливать зависимости
install [pckname] [flags]
#Удаление
remove [pckname]
#Загрузка данных о пакете в память без его установки
load [pckname]
#Выгрузка данных о пакете из памяти. Если пакет установлен, он НЕ удаляется
unload [pckname]
#Вход в отдельный поток
apistream
#Изменение rootdir, места куда будут устанавливатся пакеты
setroot [path]
#Изменение package directory
setpckdir [path]
#Получение списка пакетов
getpacks
#Изменение конфига. Параметры из нового конфига перезапишут параметры старого конфига
setconfig [path]
#Остановка сервера
stop
